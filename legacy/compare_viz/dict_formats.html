공통 template
0차: <node>.to_json()

       "result": { "type": COMM|DIFF, "score": "n/total",
                   "category": { <prop>: {"type","score","category":{…}}   // 내부노드
                                        | {"type","comp1","comp2"} } } }    // 잎(값)
2차: { "id": {"id1": <1차 id>, "id2": <1차 id>},
       "result": { "type": COMM|DIFF, "score": "n/total",
                   "category": { <prop>: {"type","score","category":{…}}   // 내부노드(재귀)
